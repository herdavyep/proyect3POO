						                for (Producto producto:listado) {

						                    if (selectedProducto == producto.getId()){

						                        int opcion = JOptionPane.showConfirmDialog(null,"Â¿Esta seguro de que desea eliminar el producto " + producto.getNombre() + "?");

						                        listarInventario.setVisible(false);
						                        ListarInventario.main();
						                    }


						                }

						            }
						        });
